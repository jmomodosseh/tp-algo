Exercice 9


Algorithme tri_croissant ;
Var
	Tab(1:5) : Tableau d’entiers ;
	min, j, i, permute, position : Entiers ;
Debut 
Pour i→ 1 à 5 faire
	min←Tab[i] ;
	position←i ;
	Pour j→ 1 à 5 faire
		Si Tab[j]<min alors
			min←Tab[j] ;
			position←j ;
		FinSi
	FinPour
	permute←Tab[i] ;
	Tab[i]←min ;
	Tab[position]←permute ;
FinPour
Fin

	FinSi

Jusqu’à i>4
Fin

Exercice 12



Algorithme recherche_dichotomique ;
Var
	Tab(1:10) : tableau d’entiers ;
	m, n : entiers ;
Debut
	Afficher "Veuillez saisir le nombre recherché" ;
	Saisir n ;
	m← 11 DIV 2 ;
Si Tab[m]=n alors,
	Afficher n, "se retrouve à la ",m, "ième position" ;
Sinon si Tab[m]>n alors
	Repeter
		m← (1+m)DIV2 ;
	Jusqu’à Tab[m]=n ;
	Afficher n, "se retrouve à la ",m, "ième position" ;
Sinon si Tab[m]<n alors
	Repeter
		m←(m+10)DIV2 ;
	Jusqu’à Tab[m]=n
	Afficher n, "se retrouve à la ",m, "ième position" ;
Sinon
	Afficher n, "ne se trouve pas dans le tableau" ;
FinSi
Fin
